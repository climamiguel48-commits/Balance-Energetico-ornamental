library(readxl)
library(dplyr)
library(tidyr)
library(ggplot2)
library(lubridate)

# -----------------------------------------------------------------------------
# Resultados preliminares - versión estable solo con ggplot2
# -----------------------------------------------------------------------------

archivos <- list(
  "CAE1" = "data/datos_horario_LE_corregidos_final_CAE1.xlsx",
  "CAE2" = "data/datos_horario_LE_corregidos_final_CAE2.xlsx",
  "Exótico" = "data/datos_horario_LE_corregidos_final_exotico.xlsx"
)

normalizar_fecha_hora <- function(x) {
  if (inherits(x, "POSIXct") || inherits(x, "POSIXt")) return(as.POSIXct(x, tz = "America/Santiago"))
  if (inherits(x, "Date")) return(as.POSIXct(x, tz = "America/Santiago"))
  if (is.numeric(x)) return(as.POSIXct(x * 86400, origin = "1899-12-30", tz = "America/Santiago"))

  suppressWarnings(parse_date_time(
    as.character(x),
    orders = c("ymd HMS", "ymd HM", "ymd", "dmy HMS", "dmy HM", "dmy", "mdy HMS", "mdy HM", "mdy"),
    tz = "America/Santiago",
    quiet = TRUE
  ))
}

asegurar_columna <- function(df, nombre) {
  if (!nombre %in% names(df)) df[[nombre]] <- NA_real_
  df
}

leer_parcela <- function(ruta, grupo) {
  df <- read_excel(ruta)

  # Homologación de nombres conocidos entre archivos
  if (!"SWC_40CM" %in% names(df) && "SWC_40CM_Avg" %in% names(df)) {
    df$SWC_40CM <- df$SWC_40CM_Avg
  }

  if (!"HR" %in% names(df) && "RH" %in% names(df)) {
    df$HR <- df$RH
  }

  if (!"temp_superficie" %in% names(df) && "TT_C" %in% names(df)) {
    df$temp_superficie <- df$TT_C
  }

  # Columnas esperadas para los gráficos
  variables_necesarias <- c(
    "temp_superficie", "velocidad_viento", "WindDir", "temp_aire", "HR", "G",
    "humedad_volumetrica", "SWC_40CM", "LE_final", "H_final", "ET_mm_h"
  )

  for (v in variables_necesarias) df <- asegurar_columna(df, v)

  # Fecha/hora principal
  if ("fecha_hora" %in% names(df)) {
    fecha_hora <- normalizar_fecha_hora(df$fecha_hora)
  } else if ("fecha" %in% names(df)) {
    fecha_hora <- normalizar_fecha_hora(df$fecha)
  } else if ("fecha.x" %in% names(df)) {
    fecha_hora <- normalizar_fecha_hora(df$fecha.x)
  } else {
    fecha_hora <- as.POSIXct(NA)
  }

  df |>
    mutate(
      grupo = grupo,
      fecha_hora = fecha_hora,
      fecha = as.Date(fecha_hora)
    ) |>
    mutate(across(all_of(variables_necesarias), ~ suppressWarnings(as.numeric(.x)))) |>
    select(grupo, fecha_hora, fecha, all_of(variables_necesarias))
}

datos <- bind_rows(lapply(names(archivos), function(g) leer_parcela(archivos[[g]], g))) |>
  filter(!is.na(fecha_hora)) |>
  arrange(grupo, fecha_hora)

variables_graficar <- tibble::tribble(
  ~variable, ~titulo, ~ylab, ~agregacion,
  "temp_superficie", "Temperatura superficial", "Temperatura superficial (°C)", "mean",
  "velocidad_viento", "Velocidad del viento", "Velocidad del viento (m s⁻¹)", "mean",
  "WindDir", "Dirección del viento", "Dirección del viento (°)", "mean",
  "temp_aire", "Temperatura del aire a 2 m", "Temperatura del aire a 2 m (°C)", "mean",
  "HR", "Humedad relativa del aire", "Humedad relativa (%)", "mean",
  "G", "Flujo de calor del suelo", "G (W m⁻²)", "mean",
  "humedad_volumetrica", "Humedad del suelo a 5 cm", "Humedad volumétrica a 5 cm (m³ m⁻³)", "mean",
  "SWC_40CM", "Humedad del suelo a 40 cm", "Humedad volumétrica a 40 cm (m³ m⁻³)", "mean",
  "LE_final", "Flujo de calor latente", "LE (W m⁻²)", "mean",
  "H_final", "Flujo de calor sensible", "H (W m⁻²)", "mean",
  "ET_mm_h", "Evapotranspiración horaria", "ET (mm h⁻¹)", "sum"
)

resumen_disponibilidad <- variables_graficar |>
  rowwise() |>
  mutate(
    registros_validos = sum(!is.na(datos[[variable]])),
    parcelas_con_datos = paste(unique(datos$grupo[!is.na(datos[[variable]])]), collapse = ", ")
  ) |>
  ungroup()

resumen_general <- datos |>
  summarise(
    registros = n(),
    fecha_inicial = min(fecha, na.rm = TRUE),
    fecha_final = max(fecha, na.rm = TRUE),
    parcelas = n_distinct(grupo)
  )

preparar_diario <- function(df, variable, agregacion = "mean") {
  if (!variable %in% names(df)) {
    return(tibble(grupo = character(), fecha = as.Date(character()), valor = numeric()))
  }

  if (agregacion == "sum") {
    salida <- df |>
      group_by(grupo, fecha) |>
      summarise(valor = sum(.data[[variable]], na.rm = TRUE), .groups = "drop")
  } else {
    salida <- df |>
      group_by(grupo, fecha) |>
      summarise(valor = mean(.data[[variable]], na.rm = TRUE), .groups = "drop")
  }

  salida |>
    mutate(valor = ifelse(is.infinite(valor) | is.nan(valor), NA_real_, valor)) |>
    filter(!is.na(fecha), !is.na(valor))
}

hay_datos <- function(df) is.data.frame(df) && nrow(df) > 0 && sum(!is.na(df$valor)) > 0

plot_variable <- function(variable, titulo, ylab, agregacion = "mean") {
  df <- preparar_diario(datos, variable, agregacion)

  if (!hay_datos(df)) {
    plot.new()
    text(0.5, 0.5, paste("Aún no hay datos suficientes para graficar:", titulo), cex = 1.1)
    return(invisible(NULL))
  }

  ggplot(df, aes(x = fecha, y = valor, color = grupo, group = grupo)) +
    geom_line(linewidth = 0.65, alpha = 0.9) +
    labs(
      title = titulo,
      x = "Fecha",
      y = ylab,
      color = "Parcela"
    ) +
    theme_minimal(base_size = 13) +
    theme(
      plot.title = element_text(face = "bold", size = 15),
      legend.position = "bottom"
    )
}
