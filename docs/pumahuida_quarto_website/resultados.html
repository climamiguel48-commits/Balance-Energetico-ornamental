p_le <- diario |>
  filter(!is.na(LE_medio)) |>
  ggplot(aes(fecha, LE_medio, color = grupo,
             text = paste("Parcela:", grupo, "<br>Fecha:", fecha, "<br>LE:", round(LE_medio, 2), "W m-2"))) +
  geom_line(linewidth = 0.75, alpha = 0.9) +
  labs(x = NULL, y = "LE medio diario (W m⁻²)", color = "Parcela") +
  theme_minimal(base_size = 13)

Precipitación y ETo

Code 
agua <- diario |>
  group_by(fecha) |>
  summarise(lluvia = max(lluvia, na.rm = TRUE), ETo = mean(ETo, na.rm = TRUE), .groups = "drop") |>
  mutate(across(where(is.numeric), ~ ifelse(is.infinite(.x) | is.nan(.x), NA_real_, .x))) |>
  pivot_longer(c(lluvia, ETo), names_to = "variable", values_to = "valor") |>
  filter(!is.na(valor))

p_agua <- ggplot(agua, aes(fecha, valor, color = variable)) +
